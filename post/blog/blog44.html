职工管理系统c++实现

    #include<iostream>
    #include<string.h>
    #define MAX 1000
    using namespace std;
    struct Person
    {
        string m_Name;
        int m_Sex;//性别，1男，2女
        int m_Age;//年龄
        string m_Phone;
        string m_Addr;
    };
    struct Addressbooks
    {
        struct Person personarray[MAX];
        int m_size;//职工人数
    };



    //函数声明
    Addressbooks* addPerson(Addressbooks *abs);
    void showPerson(Addressbooks *abs);
    int isExist(Addressbooks *abs,string name);
    void deletePerson(Addressbooks *abs);
    void findPerson(Addressbooks * abs);
    void modifyPerson(Addressbooks *abs);
    void cleanPerson(Addressbooks * abs);



    //显示菜单
    void showMenu()
    {
        cout<<"*****************"<<endl;
        cout<<"***1、添加职工***"<<endl;
        cout<<"***2、显示职工***"<<endl;
        cout<<"***3、删除职工***"<<endl;
        cout<<"***4、查找职工***"<<endl;
        cout<<"***5、修改职工***"<<endl;
        cout<<"***6、清空职工***"<<endl;
        cout<<"***7、退出系统***"<<endl;
        cout<<"*****************"<<endl;
    }
    int main(void)
    {
        Addressbooks* abs;
        abs=(struct Addressbooks*)malloc(sizeof(struct Addressbooks));
        abs->m_size=0;
        int select = 0;

        while (true)
        {
            showMenu();

            cin >> select;
            
            switch (select)
            {
            case 1:  //添加联系人
                abs=addPerson(abs);
                break;
            case 2:  //显示联系人
                showPerson(abs);
                break;
            case 3:  //删除联系人
                deletePerson(abs);
                break;
            case 4:  //查找联系人
                findPerson(abs);
                break;
            case 5:  //修改联系人
                modifyPerson(abs);
                break;
            case 6:  //清空联系人
                cleanPerson(abs);
                break;
            case 0:  //退出通讯录
                cout << "再见！" << endl;
                return 0;
                break;
            default:
                break;
            }
        }
    }
    //添加职工
    Addressbooks* addPerson(Addressbooks *abs)
    {
        if(abs->m_size==MAX)
        {
            cout<<"存储空间已满，容不下更多职工，请裁员！！！！"<<endl;
            return NULL;
        }
        else
        {
            //姓名
            string name;
            cout<<"请输入姓名"<<endl;
            cin>>name;
            abs->personarray[abs->m_size].m_Name=name;
            //性别
            cout<<"请输入性别！1->男，2->女"<<endl;
            int sex=0;
            while(true)
            {
                cin>>sex;
                if(sex==1||sex==2)
                {
                    abs->personarray[abs->m_size].m_Sex=sex;
                    break;
                }
                cout<<"输入不正确，没有第三性别，请重新输入"<<endl;
            }
            //年龄
            cout<<"请输入年龄"<<endl;
            int age=0;
            cin>>age;
            abs->personarray[abs->m_size].m_Age=age;
            //联系电话
            cout<<"请输入该职工电话"<<endl;
            string phonenumber="";
            cin>>phonenumber;
            abs->personarray[abs->m_size].m_Phone=phonenumber;
            //家庭住址
            cout<<"请输入职工住址"<<endl;
            string address="";
            cin>>address;
            abs->personarray[abs->m_size].m_Addr=address;
            //更新通讯录人数
            abs->m_size=(abs->m_size)+1;
            cout<<"添加成功"<<endl;
        }
        return abs;
    }

    //显示所有的职工信息
    void showPerson(Addressbooks *abs)
    {
        if((abs->m_size)==0)
        {
            cout<<"当前系统为空"<<endl;
        }
        else
        {
            for(int i=0;i<abs->m_size;i++)
            {
                cout << "姓名：" << abs->personarray[i].m_Name << "\t";
                cout << "性别：" << abs->personarray[i].m_Sex << "\t";
                cout << "年龄：" << abs->personarray[i].m_Age << "\t";
                cout << "电话：" << abs->personarray[i].m_Phone << "\t";
                cout << "住址：" << abs->personarray[i].m_Addr << endl;
            }
        }
        return;
    }
    //检查职工是否存在,如果存在，返回其位置，如果不存在，返回-1
    int isExist(Addressbooks *abs,string name)
    {
        for(int i=0;i<abs->m_size;i++)
        {
            if(abs->personarray[i].m_Name==name)
            {
                return i;
            }
        }
        return -1;
    }

    //删除联系人
    void deletePerson(Addressbooks *abs)
    {
        cout<<"请输入你要删除的职工的名字:"<<endl;
        string name="";
        cin>>name;
        int ret=isExist(abs,name);
        if(ret!=-1)
        {
            for(int i=ret;i<abs->m_size;i++)
            {
                abs->personarray[i]=abs->personarray[i+1];
            }
            abs->m_size--;
            cout<<"删除成功！已经没有此职工！"<<endl;
        }
        else
        {
            cout<<"没有此职工!!"<<endl;
        }
        return;
    }
    //查找职工
    void findPerson(Addressbooks * abs)
    {
        cout<<"请输入要查找的职工的名字："<<endl;
        string name="";
        cin>>name;
        int ret=isExist(abs,name);
        if(ret!=-1)
        {
            cout << "姓名：" << abs->personarray[ret].m_Name << "\t";
            cout << "性别：" << abs->personarray[ret].m_Sex << "\t";
            cout << "年龄：" << abs->personarray[ret].m_Age << "\t";
            cout << "电话：" << abs->personarray[ret].m_Phone << "\t";
            cout << "住址：" << abs->personarray[ret].m_Addr << endl;
        }
        else
        {
            cout<<"没有找到此职工！"<<endl;
        }
        return;
    }

    //修改职工信息
    void modifyPerson(Addressbooks *abs)
    {
        cout<<"请输入你要修改信息的职工的名字："<<endl;
        string name="";
        cin>>name;
        int ret=isExist(abs,name);
        if(ret!=-1)
        {
            string name1="";
            cout<<"请输入姓名："<<endl;
            cin>>name;
            abs->personarray[ret].m_Name=name;
            cout<<"请输入性别：1->男，2->女"<<endl;
            int sex=0;
            while(true)
            {
                cin>>sex;
                if(sex==1||sex==2)
                {
                    abs->personarray[ret].m_Sex=sex;
                    break;
                }
                else
                {
                    cout<<"输入错误，请重新输入"<<endl;
                }
            }
            int age=0;
            cout<<"请输入年龄："<<endl;
            cin>>age;
            abs->personarray[ret].m_Age=age;

            cout<<"请输入电话："<<endl;
            string phone = "";
            cin >> phone;
            abs->personarray[ret].m_Phone = phone;

            //家庭住址
            cout << "请输入家庭住址：" << endl;
            string address;
            cin >> address;
            abs->personarray[ret].m_Addr = address;
            cout<<"修改成功！"<<endl;
        }
        else
        {
            cout<<"没有此职工！"<<endl;
        }
        return;
    }
    //清空所有职工
    void cleanPerson(Addressbooks * abs)
    {
        abs->m_size = 0;
        cout << "通讯录已清空" << endl;
        system("pause");
        system("cls");
    }
